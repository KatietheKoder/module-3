1. What is a hash table?
A hash table is a data structure that is used to store keys/value pairs.

2. What is hashing?
Hashing is a technique that is used to uniquely identify a
specific object from a group of similar objects.

3. How does a hash table store data?
It uses a hash function to compute an index into an array in
which an element will be inserted or searched

4. How are hash tables and objects different?
Objects are used when the number of data entries are static. Hash tables
are used when data entries are variable.

7. Build a system that allows a store owner to track their store's
inventory using a hash table for storage.

class StoreInventory {
 constructor(inventory, inventoryLimit){
   this.inventory = [];
   this.inventoryLimit = 100;
 }

 hash(string, max){
 let hash = 0;
 for (let i =0; i < string.length; i++) {
   hash += string.charCodeAt(i);
 }
   return hash % max;
 }

 add(key, value) {
   let index = this.hash(key, this.inventoryLimit);
   if (this.inventory[index] === undefined) {
     this.inventory[index] = [[key, value]];
   } else {
     let inserted = false;
     for (let i = 0; i < this.inventory[index].length; i++) {
       if (this.inventory[index][i][0] === key) {
         this.inventory[index][i][1] = value;
         inserted = true;
       }
     }
     if (inserted === false) {
       this.inventory[index].push([key, value]);
     }
   }
 }

  remove(key) {
   let index = this.hash(key, this.inventoryLimit);
   if (this.inventory[index].length === 1 && this.inventory[index][0][0] === key) {
     delete this.inventory[index];
   } else {
     for (var i = 0; i < this.inventory[index]; i++) {
       if (this.inventory[index][i][0] === key) {
         delete this.inventory[index][i];
       }
     }
   }
 }

 lookup(key) {
   let index = this.hash(key, this.inventoryLimit);
   if (this.inventory[index] === undefined) {
     console.log("Item not found. Add new item?");
   } else {
     for (let i = 0; i < this.inventory[index].length; i++) {
       if (this.inventory[index][i][0] === key) {
         console.log("Total " + this.inventory[index][i][0] + "s:" + " " + this.inventory[index][i][1]);
       }
     }
   }
 }

 print() {
   console.log(this.inventory);
 }

}
