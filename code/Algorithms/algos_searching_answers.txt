Programming Assignment:

You and a friend have set a wager to see who can find the word
"Albatross" in the dictionary the fastest. Write a program to allow
you to win the bet.

function binarySearch(array, word){
  let left = 0;
  let right = array.length - 1;
  if(word < array[left] || word > array[right]) {
    return false;
  }
  while (left <= right){
    let middle = Math.floor((left + right)/2){
      if(array[middle] === word) {
       return middle;
      } else if(array[middle] < word){
        left = middle + 1;
      } else {
        right = middle - 1;
      }
    }
  }
  return null;
}

let dictionary = ["Actually", "Age", "Albatross", "Brian", "Cat"];
binarySearch(dictionary, "Albatross")

You work at a pet store, and a child has asked you to net the only
white-spotted goldfish from the fish tank. Write a program that will help
you net the right fish.

function rightFish(array, myFish) {
  for(let i = 0; i < array.length; i++){
    if(array[i] === myFish){
      return myFish;
    }
  }
  return null;
}

let fishes = ['goldfish', 'Nemo', 'white-spotted goldfish', 'Dory']
rightFish(fishes, 'white-spotted goldfish')
